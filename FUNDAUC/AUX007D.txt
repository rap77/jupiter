Declare
	MontoF     Number;
	Difer    	 Number;
Begin
  If :Dc_Numero is Null Then
	  Clear_Record;
  End If;
	  
  If :TComp is not Null Then
        --Crear_Comprobante;
        Null;
  End If;
  Go_Block('Documentos');
  First_Record;
  MontoF := 0;
  Loop
  	MontoF := MontoF + :Dc_Monto;
  	If :System.Last_Record = 'TRUE' Then 
  		 Exit;
  	End If;
    Next_Record;
  End Loop;
  If :MontoCtrl is Null Then
  	 :MontoCtrl := MontoF;
  End If;
  Difer := MontoF - :MontoCtrl;
  If Difer <> 0 Then
    Mensajes(72,1,0);
  End If;  
  Commit;
  If :Modulo is not Null Then 
     Imprimir;
  End If;
  Go_Block('Documentos');
  Clear_Block(No_Validate);
  :MontoCtrl := Null;
  Go_Item('CliPro');

-- This is a generic button procedure. It reads the NAME of the button and 
-- performs a DO_KEY(item_name).
-- Regarding QUERY-operation:
-- if there are buttons called EXECUTE_QUERY and CANCEL_QUERY, this function shows
-- them, when the ENTER_QUERY button is pressed and hides them, when EXECUTE_- or 
-- CANCEL_QUERY ispressed. No error should be returned, if these buttons do not exist.
-- To make naming of the buttons easier, EXIT, QUIT and EXIT_FORM all perform exit_form,
-- even if the form is in ENTER-QUERY mode !!!!!!!
-- A CANCEL_QUERY button-name cancels a currently "open" query.

PROCEDURE button_proc IS
  action varchar(80);

  -- hide a button (for ENTER_-, EXECUTE_- or CANCEL_QUERY)
  PROCEDURE show_off(item_name VARCHAR2) IS
  BEGIN
    IF NOT Id_Null(Find_Item(item_name)) THEN
      Set_Item_Property(item_name, DISPLAYED,PROPERTY_FALSE);
    END IF;
  END;

  -- try to show a button (for the two query-states. If there is no button to show,
  -- return FALSE to the calling procedure.
  FUNCTION show_on(item_name VARCHAR2) RETURN BOOLEAN IS
  BEGIN
    IF NOT Id_Null(Find_Item(item_name)) THEN
      Set_Item_Property(item_name,DISPLAYED,PROPERTY_TRUE);
      Set_Item_Property(item_name,ENABLED,PROPERTY_TRUE);
      RETURN TRUE;
    ELSE
      RETURN FALSE;
    END IF;
  RETURN NULL; END;

BEGIN    
  action := Get_Item_Property(NAME_IN('SYSTEM.TRIGGER_ITEM'),ITEM_NAME);
  IF action = 'ENTER_QUERY' THEN
    -- only hide the ENTER_QUERY button, if there are EXECUTE_QUERY and (!!) CANCEL_QUERY
    -- buttons. Otherwise an existing EXECUTE_QUERY button will be shown.
    IF show_on('EXECUTE_QUERY') AND show_on('CANCEL_QUERY') THEN
      show_off('ENTER_QUERY');
    END IF;
  END IF;
  IF action IN ('EXIT_FORM', 'EXIT','QUIT') THEN
    action := 'EXIT_FORM';
    IF NAME_IN('SYSTEM.MODE')='ENTER-QUERY' THEN
      Do_Key('EXIT_FORM');
    END IF;
  END IF;
  IF action = 'CANCEL_QUERY' THEN 
    action := 'EXIT_FORM';
  END IF;
  Do_Key(action);
  IF NAME_IN('SYSTEM.MODE') != 'ENTER-QUERY' THEN
    IF show_on('ENTER_QUERY') THEN
      show_off('EXECUTE_QUERY');
      show_off('CANCEL_QUERY');
    END IF;
  END IF;
END;

PROCEDURE Crear_Comprobante IS
   TipoCur		 Varchar2(3);
   ClaseCur    Number(3);
   Cursor C1 is    Select TCBA_Cuenta, TCBA_CrDb, TCBA_Monto
                     From Tipos_Compbtes_Auto
                    Where TCBA_Tipo     = :TComp
                      And TCBA_TipoDoc  = TipoCur
                      And TCBA_ClaseDoc = ClaseCur;
   R1          C1%Rowtype;
   Montoctl    Number := 0;
   Renglon     Number := 0;
   Cuenta      Number;
   CuentaCh    Varchar2(16);
   CrDb        VARCHAR2(1);
   NumeroComp  Number;
   CodeCorr    Number;
   Ref         Varchar2(20);
   DescripT    Varchar2(40);
   TCredito    Number  := 0;
   TDebito     Number  := 0;
   MontoRg     Number;
   Plan        Number  := 0;

Begin
   Totales;
   Select Crr_Numero,Crr_Codigo, TCB_Plan, TCB_Descrip
     Into NumeroComp, CodeCorr, Plan,DescripT
     From Correlativos,Tipos_Compbtes
    Where TCB_Correl = Crr_Codigo
      And Tcb_codigo = :TComp;

   NumeroComp := NumeroComp + 1;

   Update Correlativos
      Set Crr_Numero = NumeroComp
    Where Crr_Codigo = CodeCorr;

   Insert Into Comprobantes_Mae (Cm_Cia,Cm_Plan,Cm_Tipo,Cm_Periodo,Cm_Numero,Cm_Descrip,
                                 Cm_Fecha,Cm_UsrCre)
                Values (:Cia,Plan,:TComp,:Documentos.Dc_FecEmi,NumeroComp,DescripT,
                        :Documentos.Dc_FecEmi,:Global.Usuario);                      

--             ********************************************************          
--             ************       Documento Maestro        ************
--             ********************************************************
   TipoCur  := :Documentos.Dc_Tipo;
   ClaseCur := :Documentos.Dc_Clase;
   Open C1;
   Loop 
      Fetch C1 Into R1;
      Exit When C1%NOTFOUND;
      If R1.TCBA_Cuenta = 'AUTOB' Then
      	 Cuenta  := :AuxiBan;
      	 MontoRg := :TotalDep;
         CrDb    := R1.TCBA_CrDb; 
         Renglon := Renglon + 1; 
         Insert Into Comprobantes_Det (Cmd_Cia,Cmd_Plan,Cmd_Periodo,Cmd_Numero,
                              Cmd_Renglon, Cmd_Cuenta,Cmd_Referencia,Cmd_NRef,Cmd_Debe, 
                              Cmd_Haber,Cmd_Status)
              Values(:Cia,Plan,:Documentos.Dc_FecEmi,NumeroComp,
                     Renglon,Cuenta,Null,:Documentos.Dc_Numero,
                     Decode(CrDb,'D',MontoRg,Null),
                     Decode(CrDb,'H',MontoRg,Null),Null);              	 
      Else
      	 CrDb   := R1.TCBA_CrDb;
      	 First_Record; 
      	 Loop
      	   If R1.TCBA_Cuenta = 'AUTOP' Then
      	 	    Cuenta := :Dc_Auxiliar;
      	   Else
              Cuenta := R1.TCBA_Cuenta;
      	   End If;

           If R1.TCBA_Monto = 'M' Then
              MontoRg := :Documentos.Dc_Monto;
           Else
              If R1.TCBA_Monto = 'I' Then
                 MontoRg := Nvl(:Documentos.Dc_IVA,0);
              Else
                 MontoRg := :Documentos.Dc_Monto + Nvl(:Documentos.Dc_IVA,0);
              End If;
           End If;  
           Renglon := Renglon + 1;
           If MontoRg <> 0 Then      
              Insert Into Comprobantes_Det (Cmd_Cia,Cmd_Plan,Cmd_Periodo,Cmd_Numero,
                              Cmd_Renglon, Cmd_Cuenta,Cmd_Referencia,Cmd_NRef,Cmd_Debe, 
                              Cmd_Haber,Cmd_Status)
              Values(:Cia,Plan,:Documentos.Dc_FecEmi,NumeroComp,
                     Renglon,Cuenta,Null,:Documentos.Dc_Numero,
                     Decode(CrDb,'D',MontoRg,Null),
                     Decode(CrDb,'H',MontoRg,Null),Null);
           End If;                              
           If :System.Last_Record = 'TRUE' Then
          	  Exit;
           End If;           
           Next_Record;
         End Loop;
      End If;
   End Loop;
   Close C1;
Exception
   When No_Data_Found Then
      Mensajes(501,1,0);
   When Others Then
      Mensajes(502,1,0);
End;

    Set_Item_Property('TOOLBAR.TEXT',POSITION,
                       to_number(Get_Item_Property(item_name,X_POS)),
                       to_number(Get_Item_Property(item_name,Y_POS)) +
                       to_number(Get_Item_Property(item_name,HEIGHT)));

    csscale := 140/csscale;
  END IF;
END;

Procedure Mensajes (NMsg Number,Error_Flg Number,PMsg Number) Is
-- 
-- NMsg       = Numero de Mensaje
-- Error_Flag = 1 Error en Trigger
-- PMsg      <> 0 Mostar Otro Mensaje
--
Begin
  Declare
     mensaje        VARCHAR2(50);
  BEGIN
     Select MsgTxt Into :Control.mensaje -- Busca Mensaje de Error
     From   Errores
     Where  codigo  = 999
     And    nro_msg = NMsg;
     Bell;
    -- Message(mensaje,NO_ACKNOWLEDGE); Bell;
     If  Nvl(PMsg,0) <> 0 Then
         Select MsgTxt 
         Into :Control.mensaje       -- Busca Prox Mensaje de Error
         From   Errores              -- en Caso de Existir
         Where  codigo  = 999
         And    nro_msg = PMsg;
         Bell;
         --Message(mensaje); Bell;
     End If;
   Exception 
     When No_Data_Found Then
          :Control.Mensaje := (To_Char(NMsg)||': Error del Sistema');
          BELL;
 End;
      If Error_Flg = 1 Then
         Raise Form_Trigger_Failure;
      End If;

End;

PROCEDURE Crear_Comprobante_Old IS
   Cursor C1 is    Select TCBA_Cuenta, TCBA_CrDb, TCBA_Monto
                     From Tipos_Compbtes_Auto
                    Where TCBA_Tipo     = :TComp
                      And TCBA_TipoDoc  = :Documentos.Dc_Tipo
                      And TCBA_ClaseDoc = :Documentos.Dc_Clase;
   R1          C1%Rowtype;
   Montoctl    Number := 0;
   Renglon     Number := 0;
   Cuenta      Number;
   CrDb        VARCHAR2(1);
   NumeroComp  Number;
   CodeCorr    Number;
   Ref         Varchar2(20);
   DescripT    Varchar2(40);
   TCredito    Number  := 0;
   TDebito     Number  := 0;
   MontoRg     Number;
   Plan        Number  := 0;
Begin
   --Totales;
   Select Crr_Numero,Crr_Codigo, TCB_Plan, TCB_Descrip
     Into NumeroComp, CodeCorr, Plan,DescripT
     From Correlativos,Tipos_Compbtes
    Where TCB_Correl = Crr_Codigo
      And Tcb_codigo = :TComp;

   NumeroComp := NumeroComp + 1;

   Update Correlativos
      Set Crr_Numero = NumeroComp
    Where Crr_Codigo = CodeCorr;

   Insert Into Comprobantes_Mae (Cm_Cia,Cm_Plan,Cm_Tipo,Cm_Periodo,Cm_Numero,Cm_Descrip,
                                 Cm_Fecha,Cm_UsrCre)
                Values (:Cia,Plan,:TComp,Sysdate,NumeroComp,DescripT,
                        Sysdate,'JUPITER');
   :Mensaje := 'Comprobantes_Mae'; Pause;

--             ********************************************************          
--             ************       Documento Maestro        ************
--             ********************************************************
   Open C1;
   Loop 
      Fetch C1 Into R1;
      Exit When C1%NOTFOUND;
      Cuenta := R1.TCBA_Cuenta;
      CrDb   := R1.TCBA_CrDb;
      If R1.TCBA_Monto = 'M' Then
         MontoRg := :Documentos.Dc_Monto;
      Else
         If R1.TCBA_Monto = 'I' Then
            MontoRg := Nvl(:Documentos.Dc_IVA,0);
         Else
            MontoRg := :Documentos.Dc_Monto + Nvl(:Documentos.Dc_IVA,0);
         End If;
      End If;

      Renglon := Renglon + 1;
      If MontoRg <> 0 Then      
         Insert Into Comprobantes_Det (Cmd_Cia,Cmd_Plan,Cmd_Periodo,Cmd_Numero,
                              Cmd_Renglon, Cmd_Cuenta,Cmd_Referencia,Cmd_NRef,Cmd_Debe, 
                              Cmd_Haber,Cmd_Status)
              Values(:Cia,1,:Documentos.Dc_FecEmi,NumeroComp,
                     Renglon,Cuenta,:Documentos.NAuxi,:Documentos.Dc_Numero,
                     Decode(CrDb,'D',MontoRg,Null),
                     Decode(CrDb,'H',MontoRg,Null),Null);
          :Mensaje := 'Comprobantes_Det'; Pause;                     
      End If;
      MontoRg := 0;
   End Loop;
Exception
   When No_Data_Found Then
      Mensajes(501,1,0);
End;

PROCEDURE Buscar_Cuenta (TComp in Number, TipoD in Varchar2, ClaseD in Number,
                         Cuenta Out Number, CrDb out VARCHAR2) IS
BEGIN
   Select TCBA_Cuenta, TCBA_CrDb
     Into Cuenta, CrDb
     From Tipos_Compbtes_Auto
    Where TCBA_Tipo     = TComp
      And TCBA_TipoDoc  = TipoD
      And TCBA_ClaseDoc = ClaseD;
Exception
   When No_Data_Found Then
      Cuenta := 0; CrDb := 'X';  
END;

PROCEDURE Imprimir IS
  pl_id ParamList; 
BEGIN 
  pl_id := Get_Parameter_List('tmpdata'); 
  IF NOT Id_Null(pl_id) THEN 
    Destroy_Parameter_List( pl_id ); 
  END IF; 
  pl_id := Create_Parameter_List('tmpdata'); 
  Add_Parameter(pl_id,'NUMERO',TEXT_PARAMETER,To_Char(:Documentos.Dc_Numero));
  Add_Parameter(pl_id,'CIA',TEXT_PARAMETER,:Global.CodCia);
  Add_Parameter(pl_id,'TIPO',TEXT_PARAMETER,:Documentos.Dc_Tipo);
  Add_Parameter(pl_id,'CLASE',TEXT_PARAMETER,To_Char(:Documentos.Dc_Clase));
  Add_Parameter(pl_id,'PARAMFORM',TEXT_PARAMETER,'NO'); 
  Run_Product(REPORTS, :Global.Ruta||:Modulo, SYNCHRONOUS, RUNTIME, 
           FILESYSTEM, pl_id, NULL); 
END;
